Suggested normalization shape for open trades:

function normalizeOpenTradeForApi(trade, livePosition = null) {
  const quantity = trade.quantity ?? null;
  const entryPrice = trade.entryPrice ?? null;
  const currentPrice =
    livePosition?.currentPrice ??
    livePosition?.current_price ??
    trade.currentPrice ??
    null;

  const unrealizedPnl =
    typeof livePosition?.unrealizedPnl === "number"
      ? livePosition.unrealizedPnl
      : typeof livePosition?.unrealized_pl === "number"
      ? livePosition.unrealized_pl
      : trade.unrealizedPnl ?? null;

  const unrealizedPnlPct =
    typeof livePosition?.unrealizedPnlPct === "number"
      ? livePosition.unrealizedPnlPct
      : typeof livePosition?.unrealized_plpc === "number"
      ? livePosition.unrealized_plpc * 100
      : trade.unrealizedPnlPct ?? null;

  return {
    tradeId: trade.tradeId ?? null,
    symbol: trade.symbol ?? null,
    normalizedSymbol: trade.normalizedSymbol ?? null,
    assetClass: trade.assetClass ?? null,
    strategyName: trade.strategyName ?? null,
    quantity,
    entryPrice,
    currentPrice,
    stopLoss: trade.stopLoss ?? null,
    takeProfit: trade.takeProfit ?? null,
    riskAmount: trade.riskAmount ?? null,
    marketValue:
      typeof currentPrice === "number" && typeof quantity === "number"
        ? currentPrice * quantity
        : null,
    unrealizedPnl,
    unrealizedPnlPct,
    openedAt: trade.openedAt ?? null,
    status: trade.status ?? "open",
    orphaned: Boolean(trade.orphaned),
    metrics: trade.metrics ?? {},
  };
}
